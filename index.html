🐍 本機 Python 自動更新腳本 · Local Python Auto-Update Script

若瀏覽器因 CORS 安全限制無法直接獲取數據，可在電腦上執行以下 Python 腳本。它會自動從 HKJC 官方 API 下載最新開獎記錄，並將新一期數據追加到您的 CSV 檔案中。
If the browser is blocked by CORS security policy, run this Python script on your computer. It auto-fetches the latest draws from the official HKJC API and appends new records to your CSV file.
安裝依賴 / Install dependencies：pip install requests pandas
Run on a schedule via cron (macOS/Linux) or Task Scheduler (Windows) every Tue / Thu / Sat after 9 PM HK time.
# =====================================================
# 六合彩 CSV 自動更新腳本
# Mark Six CSV Auto-Updater
# 依賴 / Requirements: pip install requests pandas
# =====================================================

import requests
import pandas as pd
from datetime import datetime, timedelta

# ── 設定 / Config ──────────────────────────────────
CSV_PATH = "Mark_Six-2.csv"   # 修改為您的 CSV 路徑
                               # Change to your CSV file path
API_BASE = "https://bet.hkjc.com/marksix/getJSON.aspx"
# ──────────────────────────────────────────────────

def fetch_draws(start: str, end: str) -> list:
    """
    從 HKJC 官方 API 獲取指定日期範圍的開獎記錄
    Fetch draw records from HKJC API for the given date range
    日期格式 / Date format: YYYYMMDD
    """
    url = f"{API_BASE}?sd={start}&ed={end}&sb=0"
    try:
        resp = requests.get(url, timeout=10,
                            headers={"User-Agent": "Mozilla/5.0"})
        resp.raise_for_status()
        return resp.json()
    except Exception as e:
        print(f"  ⚠ 獲取失敗 / Fetch failed: {e}")
        return []

def update_csv():
    print("🎱 六合彩 CSV 自動更新 / Mark Six CSV Auto-Updater")
    print("─" * 52)

    # 讀取現有 CSV / Load existing CSV
    df = pd.read_csv(CSV_PATH, encoding="utf-8-sig")
    df["日期"] = pd.to_datetime(df["日期"])
    last_date = df["日期"].max()
    print(f"  現有數據最後日期 / Last date in CSV : {last_date.date()}")

    # 從最後一期之後開始獲取 / Fetch from day after last draw
    start = (last_date + timedelta(days=1)).strftime("%Y%m%d")
    end   = datetime.today().strftime("%Y%m%d")
    print(f"  正在獲取範圍 / Fetching range     : {start} → {end}")

    draws = fetch_draws(start, end)
    if not draws:
        print("  ✅ 已是最新，無需更新 / Already up to date.")
        return

    new_rows = []
    for d in draws:
        try:
            nums    = [int(x) for x in d["no"].split("+")]
            special = int(d["sno"])
            # API 日期格式 YYYY/MM/DD → 轉為 YYYY-MM-DD
            # API date format YYYY/MM/DD → convert to YYYY-MM-DD
            date_str = d["date"].replace("/", "-")
            row = {
                "期數"     : d["id"],
                "日期"     : date_str,
                "中獎號碼 1": nums[0],
                "2"        : nums[1],
                "3"        : nums[2],
                "4"        : nums[3],
                "5"        : nums[4],
                "6"        : nums[5],
                "特別號碼"  : special,
            }
            new_rows.append(row)
        except Exception as e:
            print(f"  ⚠ 解析失敗 / Parse error for draw {d.get('id','?')}: {e}")

    if new_rows:
        new_df = pd.DataFrame(new_rows)
        df = pd.concat([df, new_df], ignore_index=True)
        df.to_csv(CSV_PATH, index=False, encoding="utf-8-sig")
        print(f"  ✅ 已新增 {len(new_rows)} 期 / Added {len(new_rows)} new draw(s).")
        print(f"  最新一期 / Latest draw: "
              f"{new_rows[-1]['期數']}  {new_rows[-1]['日期']}")
    else:
        print("  ✅ 無新記錄 / No new records found.")


if __name__ == "__main__":
    update_csv()
